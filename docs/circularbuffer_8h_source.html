165 template <typename T>
166 unsigned int CircularBuffer<T>::size() const {
167  if (isFull()) {
168  return _capacity; // both start and end will point at the same index,
169  // but it is not empty in this case
170  } else {
171  if (_startIdx > _endIdx) {
172  return _endIdx + _capacity - _startIdx ; // e.g. in a 5 element array, startIdx = 2, endIdx = $
173  // endIdx points past element 4, which is 0
174  // hence the size in this case is 0 + 5 - 2, i.e. 3
175  } else { // startIdx is <= endIdx
176  return _endIdx - _startIdx;
177  // could be 0 for empty,
178  // Also remember, endIdx points past the top of the buffer,
179  // e.g. endIdx = 4 and startIdx = 2 means there are 2 elements at index 2 and 3
180  }
181  }
182 }

184 template <typename T>
185 bool CircularBuffer<T>::insert(T data) {
186  if (isFull()) {
187  // buffer is full, start overwriting
188  _startIdx = (_startIdx + 1) % _capacity;
189  }
190  _circularbuffer[_endIdx] = data;
191  _endIdx = (_endIdx + 1) % _capacity; // increment endIdx, endIdx will point one past the filled portion$
192  _flagEmpty = false;
193  return true;
194 }
